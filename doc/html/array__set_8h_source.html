38 /**
39  * Implementation of a cache-conscious string hash set.
40  *
41  * The set stores the strings as `const CharT*`.
42  *
43  * The size of a key string is limited to std::numeric_limits<KeySizeT>::max() - 1.
44  * That is 65 535 characters by default, but can be raised with the KeySizeT template parameter.
45  * See max_key_size() for an easy access to this limit.
46  *
47  * The number of elements in the set is limited to std::numeric_limits<IndexSizeT>::max().
48  * That is 4 294 967 296 elements, but can be raised with the IndexSizeT template parameter.
49  * See max_size() for an easy access to this limit.
50  *
51  * Iterators invalidation:
52  * - clear, operator=, reserve, rehash: always invalidate the iterators.
53  * - insert: only invalidate the iterators if there is a rehash.
54  * - erase: always invalidate the iterators.
55  * - shrink_to_fit: always invalidate the iterators.
56  */
57 template<class CharT,
58  class Hash = tsl::str_hash<CharT>,
59  class Traits = std::char_traits<CharT>,
60  bool StoreNullTerminator = true,
61  class KeySizeT = std::uint16_t,
62  class IndexSizeT = std::uint32_t>

353  friend bool operator==(const array_set& lhs, const array_set& rhs) {
354  if(lhs.size() != rhs.size()) {
355  return false;
356  }
357 
358  for(auto it = lhs.cbegin(); it != lhs.cend(); ++it) {
359  const auto it_element_rhs = rhs.find(it.key(), it.key_size());
360  if(it_element_rhs == rhs.cend() || it_element_rhs.value() != it_element_rhs.value()) {
361  return false;
362  }
363  }
364 
365  return true;
366  }
