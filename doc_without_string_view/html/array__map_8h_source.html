39 /**
40  * Implementation of a cache-conscious string hash map.
41  *
42  * The map stores the strings as `const CharT*`.
43  *
44  * The value T must be either nothrow move-constructible, copy-constuctible or both.
45  *
46  * The size of a key string is limited to std::numeric_limits<KeySizeT>::max() - 1.
47  * That is 65 535 characters by default, but can be raised with the KeySizeT template parameter.
48  * See max_key_size() for an easy access to this limit.
49  *
50  * The number of elements in the map is limited to std::numeric_limits<IndexSizeT>::max().
51  * That is 4 294 967 296 elements, but can be raised with the IndexSizeT template parameter.
52  * See max_size() for an easy access to this limit.
53  *
54  * Iterators invalidation:
55  * - clear, operator=: always invalidate the iterators.
56  * - insert, emplace, operator[]: always invalidate the iterators.
57  * - erase: always invalidate the iterators.
58  * - shrink_to_fit: always invalidate the iterators.
59  */
60 template<class CharT,
61  class T,
62  class Hash = tsl::str_hash<CharT>,
63  class KeyEqual = tsl::str_equal<CharT>,
64  bool StoreNullTerminator = true,
65  class KeySizeT = std::uint16_t,
66  class IndexSizeT = std::uint32_t,
67  class GrowthPolicy = tsl::power_of_two_growth_policy<2>>

205  void insert(InputIt first, InputIt last) {
206  if(std::is_base_of<std::forward_iterator_tag, typename std::iterator_traits<InputIt>::iterator_category>::value) {
207  reserve(std::distance(first, last));
208  }
209 
210  for(auto it = first; it != last; ++it) {
211  insert(it->first, it->second);
212  }

437  friend bool operator==(const array_map& lhs, const array_map& rhs) {
438  if(lhs.size() != rhs.size()) {
439  return false;
440  }
441 
442  for(auto it = lhs.cbegin(); it != lhs.cend(); ++it) {
443  const auto it_element_rhs = rhs.find_ks(it.key(), it.key_size());
444  if(it_element_rhs == rhs.cend() || it.value() != it_element_rhs.value()) {
445  return false;
446  }
447  }
448 
449  return true;
450  }
